documentation

Documentation stream intended for use within the gulp system.

documentation
Parameters
options (Object) output options
Name Description
options.format string either 'html', 'md', 'json', or 'docset'
options.filename string custom filename for md or json output
documentation ([Object]) your custom instance of documentation.js
Returns
stream.Transform:
Example
var documentation = require('./'),
    gulp = require('gulp');

gulp.task('documentation', function () {
  gulp.src('./index.js')
    .pipe(documentation({
      format: 'html'
    }))
    .pipe(gulp.dest('documentation'));
});

// documentation with JSON output, default filename API.md
gulp.task('documentation-json', function () {
  gulp.src('./index.js')
    .pipe(documentation({
      format: 'json'
    }))
    .pipe(gulp.dest('documentation'));
});

// documentation with markdown output, default filename API.md
gulp.task('documentation-md', function () {
  gulp.src('./index.js')
    .pipe(documentation({
      format: 'md'
    }))
    .pipe(gulp.dest('documentation'));
});


// documentation with JSON output, default filename API.md and custom Documentation instance
var documentation = require('gulp-documentation'),
    $documentation = require('documentation');


gulp.task('documentation-json', function () {
  gulp.src('./index.js')
    .pipe(documentation({
      format: 'json'
    }, $documentation))
    .pipe(gulp.dest('documentation'));
});
